apiVersion: apps/v1
kind: Deployment
metadata:
  name: nginx
  namespace: kube-system
spec:
  selector:
    matchLabels:
      app: nginx
  replicas: 2
  template:
    metadata:
      labels:
        app: nginx
    spec:
      containers:
      - name: nginx
        image: nginx:latest
        imagePullPolicy: Always
        ports:
        - containerPort: 80
        volumeMounts:
          - name: efs-pv
            mountPath: "/usr/share/nginx/html"
      volumes:
        - name: efs-pv
          persistentVolumeClaim:
            claimName: efs-claim
# 06.07. nodeSelector로 key=value에 해당하는 노드에만 배포되도록 수정
			nodeSelector:
        web: nginx
